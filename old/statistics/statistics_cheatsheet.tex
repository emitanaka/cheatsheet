\section{Univariate Distributions}
\subsection{Normal distribution}
Suppose $X\sim N(\mu, \sigma^2)$.
\begin{itemize}
\item $f(x) = \dfrac{1}{\sqrt{2\pi\sigma^2}} {\rm exp}\left(-\dfrac{1}{2\sigma^2}(x-\mu) ^2\right)$
\item $E(X)=\mu$ and $var(X) = \sigma^2$
\end{itemize}

\subsection{$t$ distribution}

\subsection{$\chi^2$ distribution}
Suppose $X \sim \chi^2_k$.
\begin{itemize}
	\item Support: $x \in (0, \infty)$ if $k=1$ otherwise $x \in [0, \infty)$
	\item $E(X) = k$ and $var(X) = 2k$
	\item $\sum_{i=1}^k Z_i^2 \sim \chi^2_k$ where $Z_i \sim N(0,1)$
	\item $\sum_{i=1}^k X_i^2 \sim \sigma^2 \chi^2_k$ where $X_i \sim N(0,\sigma^2)$	
	\item $\sum_{i=1}^k (X_i - \hat{\mu})^2 \sim \sigma^2 \chi^2_{k-1}$ where $X_i \sim N(\mu,\sigma^2)$
	\item $\frac{1}{m}\sum_{i=1}^k (X_i - \hat{\mu})^2 \sim \frac{\sigma^2}{m} \chi^2_{k-1}$ where $X_i \sim N(\mu,\sigma^2)$
\end{itemize}

\subsection{$F$ distribution}

\begin{itemize}
\item $\dfrac{X_1 /  d_1}{X_2 /  d_2} \sim F_{d_1, d_2}$ where $X_1 \sim \chi^2_{d_1}$ and $X_2 \sim \chi^2_{d_2}$
\item $\dfrac{S_1^2/d_1} {\sigma^2_1} / \dfrac{S_2^2/d_2}{\sigma^2_2}\sim F_{d_1, d_2} $
\item If $X \sim t_n$ then $X^2 \sim F_{1, n}$ and $X^{-2} \sim F_{n, 1}$ 
\end{itemize}

\section{Multivariate Distributions}
\subsection{Normal Distribution}
Suppose $\bs{X} \sim N(\bs{\mu}, \bs{\Sigma})$ with $|\bs{X}|=k$.
\begin{itemize}
\item $f(\bs{x}) = \dfrac{1}{\sqrt{(2\pi)^k|\bs{\Sigma}|}} {\rm exp}\left(-\dfrac{1}{2}(\bs{x}-\bs{\mu})^\tp\bs{\Sigma}^{-1} (\bs{x}-\bs{\mu})\right)$
\end{itemize}
\subsubsection{Special Cases}
$$\begin{bmatrix}
\bs{X}_1 \\
\bs{X}_2 
\end{bmatrix} \sim N\left(\begin{bmatrix}
\bs{\mu}_1\\
\bs{\mu}_2
\end{bmatrix},  \begin{bmatrix}
\bs{\Sigma}_{11} & \bs{\Sigma}_{12}\\
\bs{\Sigma}_{21} & \bs{\Sigma}_{22}
\end{bmatrix} \right)
 $$
\begin{itemize}
	\item $E(\bs{X}_2 | \bs{X}_1 = \bs{x_1}) = \bs{\mu}_2 + \bs{\Sigma}_{21}\bs{\Sigma}_{11}^{-1}(\bs{x}_1 - \bs{\mu}_1)$
	\item $var(\bs{X}_2 | \bs{X}_1 = \bs{x_1}) = \bs{\Sigma}_{22} - \bs{\Sigma}_{21}\bs{\Sigma}_{11}^{-1}\bs{\Sigma}_{12}$
\end{itemize}
